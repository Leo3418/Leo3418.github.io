{{/* Generates a widget for navigating pages from a paginator. Recognized parameters in .Scratch: - paginator: The paginator object */}} {{- $paginator := .Scratch.Get "paginator" }} {{- if gt $paginator.TotalPages 1 }} {{- $i18nAriaPrev := i18n "paginationAriaPrevious" }} {{- $i18nPrev := i18n "paginationPrevious" }} {{- $i18nAriaFirst := i18n "paginationAriaFirst" }} {{- $i18nAriaLast := i18n "paginationAriaLast" }} {{- $i18nAriaNext := i18n "paginationAriaNext" }} {{- $i18nNext := i18n "paginationNext" }}
    {{- if $paginator.HasPrev }}
  • {{- $i18nPrev -}}
    • {{- else }}
  • {{ $i18nPrev }}
    • {{- end }} {{- if eq $paginator.PageNumber 1 }}
  • 1
    • {{- else }}
  • 1
    • {{- end }} {{- $pageStart := 2 }} {{- if gt $paginator.PageNumber 5 }} {{- $pageStart = sub $paginator.PageNumber 2 }}
  • …
    • {{- end }} {{- $pageEnd := sub $paginator.TotalPages 1 }} {{- $pagesToEnd := sub $paginator.TotalPages $paginator.PageNumber }} {{- if gt $pagesToEnd 4 }} {{- $pageEnd = add $paginator.PageNumber 2 }} {{- end }} {{- if le $pageStart $pageEnd }} {{- range $i := seq $pageStart $pageEnd }}
  • {{- $i -}}
    • {{- end }} {{- end }} {{- if and (gt $pagesToEnd 3) (lt (add $pageEnd 1) $paginator.TotalPages) }}
  • …
    • {{- end }} {{- if eq $paginator.PageNumber $paginator.TotalPages }}
  • {{- $paginator.PageNumber -}}
    • {{- else }}
  • {{- $paginator.TotalPages -}}
    • {{- end }} {{- if $paginator.HasNext }}
  • {{- $i18nNext -}}
    • {{- else }}
  • {{ $i18nNext }}
    • {{- end }}
{{- end }}
